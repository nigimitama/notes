import matplotlib.pyplot as plt
import numpy as np
import scipy as sp

# гғҮгғјгӮҝгӮ’дҪңжҲҗ
n = 1000

from scipy.stats import multivariate_normal
mean = np.array([3, 5])
Sigma = np.array([
    [1, 0.5],
    [0.5, 2],
])
X = multivariate_normal.rvs(mean=mean, cov=Sigma, size=n, random_state=0)

import statsmodels.api as sm
X = sm.add_constant(X)

# зңҹгҒ®гғ‘гғ©гғЎгғјгӮҝ
beta = np.array([2, 3, 4])

# еқҮдёҖеҲҶж•ЈгҒ®е ҙеҗҲ
e = np.random.normal(loc=0, scale=1, size=n)
y = X @ beta + e

# й »еәҰдё»зҫ©
import statsmodels.api as sm
ols = sm.OLS(y, X).fit(cov_type="HC1")
ols.summary()

import pymc as pm
import arviz as az

model = pm.Model()
with model:
    beta0 = pm.Normal("beta0", mu=0, sigma=1)
    beta1 = pm.Normal("beta1", mu=0, sigma=1)
    beta2 = pm.Normal("beta2", mu=0, sigma=1)

    # иӘӨе·®еҲҶж•ЈгҒ«гӮӮз·ҡеҪўгғўгғҮгғ«гӮ’е…ҘгӮҢгӮӢ
    w0 = pm.Normal("w0", mu=0, sigma=1)
    w1 = pm.Normal("w1", mu=0, sigma=1)
    w2 = pm.Normal("w2", mu=0, sigma=1)
    lam = pm.math.exp(w0 + w1 * X[:, 1] + w2 * X[:, 2])
    sigma = pm.Exponential("sigma", lam=lam)  # еҲҶж•ЈгҒӘгҒ®гҒ§йқһиІ гҒ®еҲҶеёғгӮ’дҪҝгҒҶ

    # е№іеқҮеҖӨ mu
    mu = beta0 + beta1 * X[:, 1] + beta2 * X[:, 2]
    # иҰіжё¬еҖӨгӮ’гӮӮгҒӨзўәзҺҮеӨүж•°гҒҜ_obsгҒЁгҒҷгӮӢ
    y_obs = pm.Normal("y_obs", mu=mu, sigma=sigma, observed=y)

# гғўгғҮгғ«гӮ’GraphvizгҒ§иЎЁзӨә
pm.model_to_graphviz(model)

# гғҷгӮӨгӮәз·ҡеҪўеӣһеё°гғўгғҮгғ«гӮ’гӮөгғігғ—гғӘгғігӮ°
with model:
    idata = pm.sample(
        chains=2,
        tune=1000, # гғҗгғјгғігӮӨгғіжңҹй–“гҒ®гҖҒжҚЁгҒҰгӮӢгӮөгғігғ—гғ«ж•°
        draws=2000, # жҺЎз”ЁгҒҷгӮӢгӮөгғігғ—гғ«ж•°
        random_seed=0,
    )
